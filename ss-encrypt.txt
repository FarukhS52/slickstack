## SS-ENCRYPT PERFORMS BASIC NGINX CONFIG MODIFICATION REGARDLESS OF SS-INSTALL-NGINX ##
## IT WILL ALSO ATTEMPT TO REVERT TO OPENSSL IF LETS ENCRYPT CERTS ARE MISSING ##

## include SlickStack configuration ##
source /var/www/ss-config

## include SlickStack functions ##
source /var/www/ss-functions

####################################################################################################
#### SS-Encrypt: MESSAGE (Begin) ###################################################################
####################################################################################################

## since ss-encrypt runs every few weeks (or months) we avoid aggressive reinstall here ##
## for a cleaner SSL reinstall run ss-install or ss-update instead of ss-encrypt ##

## MESSAGE ##
echo -e "\e[36mInstalling (or renewing) free SSL certs from OpenSSL and Certbot...\e[0m"

####################################################################################################
#### SS-Encrypt: Install Nginx Config Files (Conditional) ##########################################
####################################################################################################

## here we reinstall the configuration files for Nginx in case critical lines are missing ##
## this ensures that compatible (current) Nginx boilerplates are already installed ##

found_new=$(cat /etc/nginx/nginx.conf | grep -c "/var/www/certs/")
found_lets=$(cat /etc/nginx/nginx.conf | grep -c "/var/www/certs/fullchain.pem")

## run ss-install-nginx-config (if outdated or Lets Encrypt is broken) ##
if [[ $found_new -eq 0 ]] || [[ $found_lets -gt 0 && ! -f  "/var/www/certs/fullchain.pem" ]]; then
    source /var/www/ss-install-nginx-config
fi

found_ss=$(cat /etc/nginx/nginx.conf | grep -c "slickstack.crt")

if [[ $found_ss -eq 0 ]]; then
    source /var/www/ss-install-nginx-config
fi

####################################################################################################
#### SS-Encrypt: Ensure Critical Directories Exist #################################################
####################################################################################################

## make dirs ##
mkdir /var/www/certs
mkdir /var/www/certs/keys
mkdir /var/www/html/.well-known
mkdir /var/www/html/.well-known/acme-challenge

## reset permissions (ensure that SFTP user can access if needed for e.g. Stripe, etc) ##
chown -R $SFTP_USER:www-data /var/www/html/.well-known
chmod 0775 /var/www/html/.well-known

####################################################################################################
#### SS-Encrypt: Hide Irrelevant Errors (Bug) In OpenSSL 1.1.1x ####################################
####################################################################################################

## hide error message (bug) in OpenSSL 1.1.1x (remove this snippet later) ##
touch /home/${SUDO_USER}/.rnd
touch /home/${SFTP_USER}/.rnd

####################################################################################################
#### SS-Encrypt: Regenerate Self-Signed OpenSSL Certificate ########################################
####################################################################################################

## this will generate a new self-signed OpenSSL certificate every single time it runs ##
## there are no rate-limits and it pairs perfectly with CloudFlare SSL proxy ##

## generate self-signed certificate ##
    rm /var/www/certs/*.crt
    rm /var/www/certs/keys/*.key
    ## for all other domains e.g. example.com or www.example.com ##
    openssl req -new -x509 -nodes -days 3650 -newkey rsa:2048 -keyout /var/www/certs/keys/slickstack.key -out /var/www/certs/slickstack.crt -subj "/C=${CSR_COUNTRY}/ST=${CSR_STATE}/L=${CSR_CITY}/O=${CSR_COMPANY}/OU=${CSR_DEPT}/CN=${SITE_DOMAIN}"

####################################################################################################
#### SS-Encrypt: (Re)configure Nginx (Prepare For Certbot Verification) ############################
####################################################################################################

## here we need to check if:
# Lets Encrypt certs do not exist (or wrong based on current ss-config multisite/domain) then
# revert Nginx config to OpenSSL now to allow for Certbot verify domain again
# or if Lets Encrypt 

## restart Nginx ##
/etc/init.d/nginx restart

####################################################################################################
#### SS-Encrypt: Generate DH Parameters (If Lets Encrypt Enabled) ##################################
####################################################################################################

## this will generate DH parameters using OpenSSL for use by Lets Encrypt certificates ##
## by maintaining this file we are able to strengthen the security of the SSL ##

if [[ "$SSL_TYPE" == "certbot" ]] && [[ ! -f "/var/www/certs/dhparam.pem" ]]; then 
    openssl dhparam -out /var/www/certs/dhparam.pem 2048
fi

####################################################################################################

#### SS-Encrypt: Request Certificates From Lets Encrypt Servers (Rate-Limited) #####################
####################################################################################################

## NOTICE: SLICKSTACK DOES NOT SUPPORT SUB-SUBDOMAINS LIKE WWW.BLOG.EXAMPLE.COM ##

## SlickStack deletes old Lets Encrypt certs and requests new ones to avoid dir-stacking ##
## this approach also allows you to change your server to a different CMS or TLD ##

## SINGLE SITE CERTS + STAGING/DEV BOTH ENABLED ##

## if site domain contains www (and is therefore a subdomain) e.g. www.example.com or www.example.co.uk ##
if [[ "$SITE_DOMAIN" == www.* ]] && [[ "$STAGING_SITE" != "false" ]] && [[ "$STAGING_SITE_SUBDOMAIN" != "false" ]] && [[ "$DEV_SITE" != "false" ]] && [[ "$DEV_SITE_SUBDOMAIN" != "false" ]]; then
    certbot delete --cert-name ${SITE_TLD}
    certbot delete --cert-name ${SITE_DOMAIN}
    certbot certonly --noninteractive --agree-tos --cert-name ${SITE_TLD} -d ${SITE_TLD} -d www.${SITE_TLD} -d dev.${SITE_TLD} -d staging.${SITE_TLD} --register-unsafely-without-email --webroot -w /var/www/html/

## if site domain does not contain www (and is not any other type of subdomain) e.g. example.com or example.co.uk ##
elif [[ "$SITE_DOMAIN" != www.* ]] && [[ "$SITE_TLD" == "$SITE_DOMAIN" ]] && [[ "$STAGING_SITE" != "false" ]] && [[ "$STAGING_SITE_SUBDOMAIN" != "false" ]] && [[ "$DEV_SITE" != "false" ]] && [[ "$DEV_SITE_SUBDOMAIN" != "false" ]]; then
    certbot delete --cert-name ${SITE_TLD}
    certbot delete --cert-name ${SITE_DOMAIN}
    certbot certonly --noninteractive --agree-tos --cert-name ${SITE_TLD} -d ${SITE_TLD} -d www.${SITE_TLD} -d dev.${SITE_TLD} -d staging.${SITE_TLD} --register-unsafely-without-email --webroot -w /var/www/html/

## if site domain is a non-www type of subdomain e.g. blog.example.com or blog.example.co.uk ##
elif [[ "$SITE_DOMAIN" != www.* ]] && [[ "$SITE_TLD" != "$SITE_DOMAIN" ]] && [[ "$STAGING_SITE" != "false" ]] && [[ "$STAGING_SITE_SUBDOMAIN" != "false" ]] && [[ "$DEV_SITE" != "false" ]] && [[ "$DEV_SITE_SUBDOMAIN" != "false" ]]; then
    certbot delete --cert-name ${SITE_TLD}
    certbot delete --cert-name ${SITE_DOMAIN}
    certbot certonly --noninteractive --agree-tos --cert-name ${SITE_DOMAIN} -d ${SITE_DOMAIN} -d dev.${SITE_TLD} -d staging.${SITE_TLD} --register-unsafely-without-email --webroot -w /var/www/html/


## SINGLE SITE CERTS + STAGING ENABLED / DEV DISABLED ##

## if site domain contains www (and is therefore a subdomain) e.g. www.example.com or www.example.co.uk ##
elif [[ "$SITE_DOMAIN" == www.* ]] && [[ "$STAGING_SITE" != "false" ]] && [[ "$STAGING_SITE_SUBDOMAIN" != "false" ]] && [[ "$DEV_SITE" == "false" || "$DEV_SITE_SUBDOMAIN" == "false" ]]; then
    certbot delete --cert-name ${SITE_TLD}
    certbot delete --cert-name ${SITE_DOMAIN}

## if site domain does not contain www (and is not any other type of subdomain) e.g. example.com or example.co.uk ##
elif [[ "$SITE_DOMAIN" != www.* ]] && [[ "$SITE_TLD" == "$SITE_DOMAIN" ]] && [[ "$STAGING_SITE" != "false" ]] && [[ "$STAGING_SITE_SUBDOMAIN" != "false" ]] && [[ "$DEV_SITE" == "false" || "$DEV_SITE_SUBDOMAIN" == "false" ]]; then
    certbot delete --cert-name ${SITE_TLD}
    certbot delete --cert-name ${SITE_DOMAIN}
    certbot certonly --noninteractive --agree-tos --cert-name ${SITE_TLD} -d ${SITE_TLD} -d www.${SITE_TLD} -d staging.${SITE_TLD} --register-unsafely-without-email --webroot -w /var/www/html/

## if site domain is a non-www type of subdomain e.g. blog.example.com or blog.example.co.uk ##
elif [[ "$SITE_DOMAIN" != www.* ]] && [[ "$SITE_TLD" != "$SITE_DOMAIN" ]] && [[ "$STAGING_SITE" != "false" ]] && [[ "$STAGING_SITE_SUBDOMAIN" != "false" ]] && [[ "$DEV_SITE" == "false" || "$DEV_SITE_SUBDOMAIN" == "false" ]]; then
    certbot delete --cert-name ${SITE_TLD}
    certbot delete --cert-name ${SITE_DOMAIN}
    certbot certonly --noninteractive --agree-tos --cert-name ${SITE_DOMAIN} -d ${SITE_DOMAIN} -d staging.${SITE_TLD} --register-unsafely-without-email --webroot -w /var/www/html/


## SINGLE SITE CERTS + STAGING DISABLED / DEV ENABLED ##

## if site domain contains www (and is therefore a subdomain) e.g. www.example.com or www.example.co.uk ##
elif [[ "$SITE_DOMAIN" == www.* ]] && [[ "$STAGING_SITE" == "false" || "$STAGING_SITE_SUBDOMAIN" == "false" ]] && [[ "$DEV_SITE" != "false" ]] && [[ "$DEV_SITE_SUBDOMAIN" != "false" ]]; then
    certbot delete --cert-name ${SITE_TLD}
    certbot delete --cert-name ${SITE_DOMAIN}
    certbot certonly --noninteractive --agree-tos --cert-name ${SITE_TLD} -d ${SITE_TLD} -d www.${SITE_TLD} -d dev.${SITE_TLD} --register-unsafely-without-email --webroot -w /var/www/html/

## if site domain does not contain www (and is not any other type of subdomain) e.g. example.com or example.co.uk ##
elif [[ "$SITE_DOMAIN" != www.* ]] && [[ "$SITE_TLD" == "$SITE_DOMAIN" ]] && [[ "$STAGING_SITE" == "false" || "$STAGING_SITE_SUBDOMAIN" == "false" ]] && [[ "$DEV_SITE" != "false" ]] && [[ "$DEV_SITE_SUBDOMAIN" != "false" ]]; then
    certbot delete --cert-name ${SITE_TLD}
    certbot delete --cert-name ${SITE_DOMAIN}
    certbot certonly --noninteractive --agree-tos --cert-name ${SITE_TLD} -d ${SITE_TLD} -d www.${SITE_TLD} -d dev.${SITE_TLD} --register-unsafely-without-email --webroot -w /var/www/html/

## if site domain is a non-www type of subdomain e.g. blog.example.com or blog.example.co.uk ##
elif [[ "$SITE_DOMAIN" != www.* ]] && [[ "$SITE_TLD" != "$SITE_DOMAIN" ]] && [[ "$STAGING_SITE" == "false" || "$STAGING_SITE_SUBDOMAIN" == "false" ]] && [[ "$DEV_SITE" != "false" ]] && [[ "$DEV_SITE_SUBDOMAIN" != "false" ]]; then
    certbot delete --cert-name ${SITE_TLD}
    certbot delete --cert-name ${SITE_DOMAIN}
    certbot certonly --noninteractive --agree-tos --cert-name ${SITE_DOMAIN} -d ${SITE_DOMAIN} -d dev.${SITE_TLD} --register-unsafely-without-email --webroot -w /var/www/html/


## SINGLE SITE CERTS + STAGING/DEV BOTH DISABLED ##

## if site domain contains www (and is therefore a subdomain) e.g. www.example.com or www.example.co.uk ##
elif [[ "$SITE_DOMAIN" == www.* ]] && [[ "$STAGING_SITE" == "false" || "$STAGING_SITE_SUBDOMAIN" == "false" ]] && [[ "$DEV_SITE" == "false" || "$DEV_SITE_SUBDOMAIN" == "false" ]]; then
    certbot delete --cert-name ${SITE_TLD}
    certbot delete --cert-name ${SITE_DOMAIN}
    certbot certonly --noninteractive --agree-tos --cert-name ${SITE_TLD} -d ${SITE_TLD} -d www.${SITE_TLD} --register-unsafely-without-email --webroot -w /var/www/html/

## if site domain does not contain www (and is not any other type of subdomain) e.g. example.com or example.co.uk ##
elif [[ "$SITE_DOMAIN" != www.* ]] && [[ "$SITE_TLD" == "$SITE_DOMAIN" ]] && [[ "$STAGING_SITE" != "false" ]] && [[ "$STAGING_SITE_SUBDOMAIN" != "false" ]] && [[ "$DEV_SITE" != "false" ]] && [[ "$DEV_SITE_SUBDOMAIN" != "false" ]]; then
    certbot delete --cert-name ${SITE_TLD}
    certbot delete --cert-name ${SITE_DOMAIN}
    certbot certonly --noninteractive --agree-tos --cert-name ${SITE_TLD} -d ${SITE_TLD} -d www.${SITE_TLD} --register-unsafely-without-email --webroot -w /var/www/html/

## if site domain is a non-www type of subdomain e.g. blog.example.com or blog.example.co.uk ##
elif [[ "$SITE_DOMAIN" != www.* ]] && [[ "$SITE_TLD" != "$SITE_DOMAIN" ]] && [[ "$STAGING_SITE" != "false" ]] && [[ "$STAGING_SITE_SUBDOMAIN" != "false" ]] && [[ "$DEV_SITE" != "false" ]] && [[ "$DEV_SITE_SUBDOMAIN" != "false" ]]; then
    certbot delete --cert-name ${SITE_TLD}
    certbot delete --cert-name ${SITE_DOMAIN}
    certbot certonly --noninteractive --agree-tos --cert-name ${SITE_DOMAIN} -d ${SITE_DOMAIN} --register-unsafely-without-email --webroot -w /var/www/html/



## otherwise just do this ##
else
#    certbot certonly --noninteractive --agree-tos --cert-name ${SITE_TLD} --expand -d ${SITE_DOMAIN} --register-unsafely-without-email --webroot -w /var/www/html/
    echo -e "\e[36mIt seems required options are missing from ss-config so we dont know what Lets Encrypt cert to request for you...\e[0m"
fi

####################################################################################################
#### SS-Encrypt: Create Lets Encrypt Symlinks In SlickStack Directories ############################
####################################################################################################

## here we create symlinks in the SlickStack directory that point to Lets Encrypt ##
## this approach simplifies our Nginx settings and makes it more stable ##

## create symlinks based on domain conditions ##
if [[ "$SITE_DOMAIN" != www.* ]] && [[ "$SITE_TLD" != "$SITE_DOMAIN" ]] && [[ -d  "/etc/letsencrypt/live/${SITE_DOMAIN}" ]]; then
    ln /etc/letsencrypt/live/${SITE_DOMAIN}/cert.pem /var/www/certs/cert.pem
    ln /etc/letsencrypt/live/${SITE_DOMAIN}/privkey.pem /var/www/certs/keys/privkey.pem
    ln /etc/letsencrypt/live/${SITE_DOMAIN}/chain.pem /var/www/certs/chain.pem
    ln /etc/letsencrypt/live/${SITE_DOMAIN}/fullchain.pem /var/www/certs/fullchain.pem
else
    ln /etc/letsencrypt/live/${SITE_TLD}/cert.pem /var/www/certs/cert.pem
    ln /etc/letsencrypt/live/${SITE_TLD}/privkey.pem /var/www/certs/keys/privkey.pem
    ln /etc/letsencrypt/live/${SITE_TLD}/chain.pem /var/www/certs/chain.pem
    ln /etc/letsencrypt/live/${SITE_TLD}/fullchain.pem /var/www/certs/fullchain.pem
fi

####################################################################################################
#### SS-Encrypt: Modify Nginx Configuration For Lets Encrypt #######################################
####################################################################################################

## here we modify the live Nginx configuration files to activate Lets Encrypt certificates ##
## this will only be completed if the Certbot process appears to be successful ##

## modify live Nginx configuration ##
if [[ "$SSL_TYPE" == "certbot" ]] && [[ -d  "/etc/letsencrypt/live" ]] && [[ -f  "/var/www/certs/fullchain.pem" ]] && [[ -f  "/var/www/certs/keys/privkey.pem" ]]; then
    sed -i "s#/var/www/certs/slickstack.crt#/var/www/certs/fullchain.pem#g" /etc/nginx/nginx.conf
    sed -i "s#/var/www/certs/keys/slickstack.key#/var/www/certs/keys/privkey.pem#g" /etc/nginx/nginx.conf
    sed -i "s/#@CERTBOT# //g" /etc/nginx/nginx.conf
    echo -e "\e[36mLets Encrypt SSL was succesfully installed...\e[0m"
else
    echo -e "\e[36mCertbot installation was not successful, reverting to OpenSSL...\e[0m"
fi

####################################################################################################
#### SS-Encrypt: Reset Permissions (OpenSSL + Certbot) #############################################
####################################################################################################

## reset permissions ##
mkdir /var/www/certs
mkdir /var/www/certs/keys
mkdir /var/www/html/.well-known
mkdir /var/www/html/.well-known/acme-challenge

chown root:root /var/www/certs ## must be root:root
chown root:root /var/www/certs/keys ## must be root:root
chown -R $SFTP_USER:www-data /var/www/html/.well-known ## must be $SFTP_USER:www-data
chmod 0755 /var/www/certs ## must be 0755
chmod 0700 /var/www/certs/keys ## must be 0700
chmod 0644 /var/www/certs/*.crt ## must be 0644
chmod 0644 /var/www/certs/*.pem ## must be 0644
chmod 0600 /var/www/certs/keys/*.key ## must be 0600
chmod 0600 /var/www/certs/keys/*.pem ## must be 0600
chmod 0775 /var/www/html/.well-known

####################################################################################################
#### SS-Encrypt: Restart Nginx (After Certbot SSL Installation Complete) ###########################
####################################################################################################

## restart nginx ##
/etc/init.d/nginx restart

####################################################################################################
#### SS-Encrypt: Touch Timestamp File (End Script) #################################################
####################################################################################################

## this is a dummy timestamp file that will remember the last time this script was run ##

touch /var/www/meta/.timestamp-ss-encrypt
