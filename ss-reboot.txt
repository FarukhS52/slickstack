#### path: /var/www/ss-reboot ######################################################################
#### destination: n/a (not a boilerplate) ##########################################################
#### purpose: Force reboots the server but is rate-limited as per associated .timestamp file #######
#### module version: Ubuntu 20.04 LTS ##############################################################
#### sourced by: ss core bash scripts, ss core cron jobs ###########################################
####################################################################################################

## include SlickStack configuration ##
source /var/www/ss-config

## include SlickStack functions ##
source /var/www/ss-functions

####################################################################################################
#### SS-Reboot: Force Reboot The Server ############################################################
####################################################################################################

## reboot only if ss-reboot was not run in past 60 minutes ##
if [[ -n $(find "/var/www/meta/.timestamp-ss-reboot" -amin +60) ]]; then
    /bin/bash -c "/sbin/reboot"
fi

####################################################################################################
#### SS-Reboot: Touch Timestamp File (End Of Script) ###############################################
####################################################################################################

## this is a dummy timestamp file that will remember the last time this script was run ##
## it can be useful for developer reference and it sometimes used by SlickStack ##

## script timestamp ##
touch /var/www/meta/.timestamp-ss-reboot
