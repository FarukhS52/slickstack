#!/bin/bash

####################################################################################################
#### author: SlickStack ############################################################################
#### link: https://slickstack.io ###################################################################
#### mirror: http://mirrors.slickstack.io/crons/09-cron-half-weekly.txt ############################
#### path: /var/www/crons/09-cron-half-weekly ######################################################
#### destination: n/a (not a boilerplate) ##########################################################
#### purpose: SlickStack core cron job *half-weekly* (09/13) will run every ~3 days ################
#### module version: Ubuntu 20.04 LTS ##############################################################
#### sourced by: root crontab ######################################################################
#### bash aliases: ss cron half weekly #############################################################
####################################################################################################

## NEVER MODIFY THE ROOT CRONTAB AND SS CORE CRON JOBS OR YOUR STACK WILL STOP WORKING ##
## INSTEAD ADJUST SS_INTERVAL SETTINGS IN SS-CONFIG OR /var/www/crons/custom/ ##

## include SlickStack configuration ##
source /var/www/ss-config

## include SlickStack functions ##
source /var/www/ss-functions

####################################################################################################
#### 09-Cron-Half-Weekly: Ensure SS-Check + SS-Worker Intact (Core Bash Scripts) ###################
####################################################################################################

## SNIPPET: ss core cron jobs

## this improves redundancy by having ss core cron jobs ensure ss core bash scripts exist ##
## the root crontab in SlickStack likewise ensures that ss core cron jobs exist ##

## ensure ss-check exists and intact ##
if [[ ! -f "$PATH_SS_CHECK" ]] || [[ -z $(grep -q "$SS_EOF" "$PATH_SS_CHECK") ]]; then
    rm /tmp/ss-check*
    wget -O /tmp/ss-check "$MIRROR_SS_CHECK"
    cp /tmp/ss-check /var/www/ss-check
    chown root:root /var/www/ss-check ## must be root:root
    chmod 0700 /var/www/ss-check ## 0700 means root can execute
    rm /tmp/ss-check*
fi

## ensure ss-worker exists and intact ##
if [[ ! -f "$PATH_SS_WORKER" ]] || [[ -z $(grep -q "$SS_EOF" "$PATH_SS_WORKER") ]]; then
    rm /tmp/ss-worker*
    wget -O /tmp/ss-worker "$MIRROR_SS_WORKER"
    cp /tmp/ss-worker /var/www/ss-worker
    chown root:root /var/www/ss-worker ## must be root:root
    chmod 0700 /var/www/ss-worker ## 0700 means root can execute
    rm /tmp/ss-worker*
fi

####################################################################################################
#### 09-Cron-Half-Weekly: Run Custom Tasks #########################################################
####################################################################################################

## this will run custom shell commands that you can save to a reserved filename below ##
## carefully consider server resources and best practices before activating ##

## run custom cron jobs ##
SS_CUSTOM_CRON_HALF_WEEKLY_NOW=$(source /var/www/ss-config; echo $SS_CUSTOM_CRON_HALF_WEEKLY)
if [[ -n "$SS_CUSTOM_CRON_HALF_WEEKLY" ]]; then 
    $SS_CUSTOM_CRON_HALF_WEEKLY_NOW
fi

## run custom cron jobs ##
if [[ -f "$PATH_CRON_HALF_WEEKLY_CUSTOM" ]]; then
    source "$PATH_CRON_HALF_WEEKLY_CUSTOM"
fi

####################################################################################################
#### 09-Cron-Half-Weekly: Run Scheduled Tasks ######################################################
####################################################################################################

## the below tasks will be called if configured to run at this interval in ss-config ##
## certain tasks are automatically called if the relevant interval is missing ##

## run ss-clean-files if set to half-weekly ##
if [[ "$SS_INTERVAL_CLEAN_FILES" == "half-weekly" ]]; then 
    source /var/www/ss-clean-files
fi

## run ss-sync-staging if set to half-weekly ##
if [[ "$SS_INTERVAL_SYNC_STAGING" == "half-weekly" ]]; then 
    source /var/www/ss-sync-staging
fi

## run ss-install-wordpress-config if set to half-weekly ##
if [[ "$SS_INTERVAL_INSTALL_WORDPRESS_CONFIG" == "half-weekly" ]]; then 
    source /var/www/ss-install-wordpress-config
fi

## run ss-install-wordpress-mu-plugins if set to half-weekly ##
if [[ "$SS_INTERVAL_INSTALL_WORDPRESS_MU_PLUGINS" == "half-weekly" ]]; then 
    source /var/www/ss-install-wordpress-mu-plugins
fi

## run ss-dos2unix if set to half-weekly or if not defined (default) ##
if [[ "$SS_INTERVAL_DOS2UNIX" == "half-weekly" || -z "$SS_INTERVAL_DOS2UNIX" ]]; then 
    source /var/www/ss-dos2unix
fi

## run ss-scan-malware if set to half-weekly ##
if [[ "$SS_INTERVAL_SCAN_MALWARE" == "half-weekly" ]]; then 
    source /var/www/ss-scan-malware
fi

## run ss-optimize-database if set to half-weekly ##
if [[ "$SS_INTERVAL_OPTIMIZE_DATABASE" == "half-weekly" ]]; then 
    source /var/www/ss-optimize-database
fi

## run ss-encrypt if set to half-weekly ##
if [[ "$SS_INTERVAL_ENCRYPT" == "half-weekly" ]]; then 
    source /var/www/ss-encrypt
fi

## run ss-purge if set to half-weekly ##
if [[ "$SS_INTERVAL_PURGE" == "half-weekly" ]]; then 
    source /var/www/ss-purge
fi

## run ss-install-ubuntu-crontab if set to half-weekly ##
if [[ "$SS_INTERVAL_INSTALL_UBUNTU_CRONTAB" == "half-weekly" ]]; then 
    source /var/www/ss-install-ubuntu-crontab
fi

## run ss-install-ubuntu-bash if set to half-weekly or if not defined (default) ##
if [[ "$SS_INTERVAL_INSTALL_UBUNTU_BASH" == "half-weekly" || -z "$SS_INTERVAL_INSTALL_UBUNTU_BASH" ]]; then 
    source /var/www/ss-install-ubuntu-bash
fi

####################################################################################################
#### 09-Cron-Half-Weekly: Reset Permissions (SS Core Bash Scripts + SS Core Cron Jobs) #############
####################################################################################################

## SNIPPET: ss-check, ss-worker, ss core cron jobs

## we hardcode this permissions reset snippet into all ss core scripts for redundancy ##
## chmod 0700 means only the root/sudo users can execute the ss core scripts ##

## reset permissions ##
chown root:root /var/www/ss* ## must be root:root
chown root:root /var/www/crons/*cron* ## must be root:root
chmod 0700 /var/www/ss* ## 0700 means root can execute
chmod 0700 /var/www/crons/*cron* ## 0700 means root can execute

####################################################################################################
#### 09-Cron-Half-Weekly: Delete Lock File #########################################################
####################################################################################################

## here we delete the lock file associated with this cron job to clear the cron queue ##
## this is technically not necessary but we do it anyway for extra security ##

## delete lock file ##
rm /tmp/09-cron-half-weekly*.lock*

####################################################################################################
#### 09-Cron-Half-Weekly: Touch Timestamp File (End Script) ########################################
####################################################################################################

## this is a dummy timestamp file that will remember the last time this script was run ##
## it can be useful for developer reference and is sometimes used by SlickStack ##

## script timestamp ##
touch /var/www/meta/.timestamp-09-cron-half-weekly
